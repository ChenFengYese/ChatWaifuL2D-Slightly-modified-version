# 运行方法

conda create --name chatWaifu python=3.10 <br>
conda activate chatWaifu <br>
pip install -r requirements.txt <br>
python ChatWaifuService.py <br>
运行ChatWaifuGameL2D中的ChatWaifuL2D.exe即可 <br>
